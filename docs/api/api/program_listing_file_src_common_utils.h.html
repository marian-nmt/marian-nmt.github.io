// prints vector values with a custom label.
template<class T>
void Debug(const T *arr, size_t size, const std::string &str) {
  std::cerr << str << ":" << size << ": ";
  for (size_t i = 0; i < size; ++i) {
    std::cerr << arr[i] << " ";
  }
  std::cerr << std::endl;
}

template<class T>
void Debug(const std::vector<T> &arr, const std::string &str) {
  Debug(arr.data(), arr.size(), str);
}
