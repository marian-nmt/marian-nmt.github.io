        PARAM_TYPE_BOOL,
        PARAM_TYPE_STRING
    };

    class RegisteredParam {
    private:
        std::string m_name;
        ParameterType m_type;
        void * m_data;

    public:
        RegisteredParam() {}

            m_type = type;
            m_data = data;
        }

        const ParameterType& Type() const {return m_type;}
        void * Data() const {return m_data;}
