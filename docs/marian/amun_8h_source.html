156  auto trgWordDrop = graph->dropout(dropoutTrg, {1, 1, trgWords});
157  embeddings = dropout(embeddings, mask=trgWordDrop);
158  }
159 
160  if(!attention_)
161  attention_ = New<GlobalAttention>("decoder",
162  state->getEncoderState(),
163  dimDecState,
164  dropout_prob=dropoutRnn,
165  normalize=layerNorm);
166 
167  if(!rnn)
168  rnn = New<RNN<CGRU>>(graph, "decoder",
169  dimTrgEmb, dimDecState,
170  attention_,
171  dropout_prob=dropoutRnn,
172  normalize=layerNorm);
173  auto stateOut = (*rnn)(embeddings, stateAmun->getState());
174 
175  bool single = stateAmun->doSingleStep();
176 
177  auto alignedContextsVec = attention_->getContexts();
178  auto alignedContext = single ?
179  alignedContextsVec.back() :
180  concatenate(alignedContextsVec, keywords::axis=2);
181 
183  auto logitsL1 = Dense("ff_logit_l1", dimTrgEmb,
184  activation=act::tanh,
185  normalize=layerNorm)
186  (embeddings, stateOut, alignedContext);
187 
188  auto logitsOut = Dense("ff_logit_l2", dimTrgVoc)(logitsL1);
189 
190  return New<DecoderStateAmun>(stateOut, logitsOut,
191  state->getEncoderState());
